const iOSLink = "https://apps.apple.com/us/app/my-app/id1234567890"; const AndroidLink = "https://play.google.com/store/apps/details?id=com.mycompany.myapp"; function getOS() { // Cihazın işletim sistemini belirle const userAgent = window.navigator.userAgent; const platform = userAgent.match(/(iPhone|iPad|iPod)/i); return platform ? "iOS" : "Android"; } function getLink() { // Kullanıcının işletim sistemini kontrol et const os = getOS(); // Doğru linki döndür return os === "iOS" ? iOSLink : AndroidLink; }
